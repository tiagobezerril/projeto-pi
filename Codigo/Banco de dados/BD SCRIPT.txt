SELECT s.idsensor as 'ID Sensor',
  da.porcentagem AS 'Porcentagem',
  da.dtdecoleta as 'Data de Coleta'
FROM sensor as s 
JOIN dados as da ON idsensor = fksensor 
WHERE idsensor = 2009;
